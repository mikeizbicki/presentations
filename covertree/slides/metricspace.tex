\begin{defn}
A \emph{metric space} is a set $\mathcal{X}$ equiped with a distance function $d : \mathcal{X} \times \mathcal{X} \rightarrow \mathbb{R}$ such that:
\begin{center}
\begin{tabular}{ll}
$d(x,y) \ge 0$ & $d(x,y) = 0$ iff $x=y$ \\
$d(x,y) = d(y,x)$ & $d(x,z) \le d(x,y) + d(y,z)$ \\
\end{tabular}
\end{center}
\end{defn}
